			$x_{dj} = \begin{cases}

				0 & \text{otherwise}\\
			\end{cases}$\\
		}
		\Else{
			$\tau_{ij}^{(vt)(d)} = \tau_{ij}^{(vt)(d)} + x_{ij} \cdot \delta \cdot \frac{|R|}{|R^{update}|}$ , $\delta = 3$\\
			$x_{ij} = \begin{cases} 
				1 & \text{if edge from $v_i$ to $v_j$ exists in $R^{update}$}\\ 
				0 & \text{otherwise}\\ 
			\end{cases}$\\
		}

\end{algorithm}
